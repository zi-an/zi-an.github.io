func splitMessage(text string, limit int) []string {
	re := regexp.MustCompile("``" + "`\n")
	if len(text) > limit && strings.Index(text, "``"+"`\n") == -1 {
		re = regexp.MustCompile("\n\n")
	}
	arr := []int{0}
	for _, loc := range re.FindAllStringIndex(text, -1) {
		arr = append(arr, loc[1])
	}
	arr = append(arr, len(text))
	var groups [][]int
	startIdx := 0
	for i := 1; i < len(arr); i++ {
		if arr[i]-arr[startIdx] > limit {
			groups = append(groups, arr[startIdx:i])
			startIdx = i - 1
		}
	}
	groups = append(groups, arr[startIdx:])
	var messages []string
	for _, g := range groups {
		fmt.Println("åˆ†:", g[0], g[len(g)-1])
		temp := text[g[0]:g[len(g)-1]]
		temp = regexp.MustCompile(`(?m)^\s*\n`).ReplaceAllString(temp, "")
