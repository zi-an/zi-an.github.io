    powersave.sh # 

    #!/bin/bash

logintime=`last docker --time-format iso | head -n1 | awk '{print $6}'`
#logintime="2024-09-11T07:18:18+08:00"
if [ ${#logintime} -eq 6 ];then
    echo "555" >> 555.log
else
        logintime=`date -d $logintime +%s`
        logintime=$(expr $logintime)
        now=$(expr `date +%s`)
        now=$(expr $now - $logintime)
        if [ $now -gt 7200 ] ;then
                systemctl suspend
        fi
fi
