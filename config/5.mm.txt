function FindProxyForURL(url, host) {
    if (
	dnsDomainIs(host, ".google.com") || dnsDomainIs(host, ".googlevideo.com") || dnsDomainIs(host, ".googleapis.com") || dnsDomainIs(host, ".gstatic.com") ||
	dnsDomainIs(host, ".youtube.com") || dnsDomainIs(host, "grok.com") || dnsDomainIs(host, ".grok.com") || dnsDomainIs(host, ".intercom.io") ||
	dnsDomainIs(host, ".wikipedia.org") || dnsDomainIs(host, ".wikimedia.org")
	) {
        return "PROXY 10.0.0.5:20172";
    }
    return "DIRECT";
}
